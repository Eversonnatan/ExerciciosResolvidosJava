06. Um funcionário recebe um salário fixo mais 4% de 
comissão sobre as vendas (4% do valor das vendas efetuadas pelo funcionário). 
Faça um programa que receba o salário fixo do funcionário e 
o valor de suas vendas, calcule 
e mostre a comissão e seu salário final.

public static void main(String[] args) {
 double sal_fixo,comiss,vendas,sal_final;
 System.out.print("Digite seu salario fixo: ");
 sal_fixo = Double.parseDouble(System.console().readLine());
 System.out.print("Digite o valor da suas vendas: ");
 vendas = Double.parseDouble(System.console().readLine());
 comiss = ((vendas*4)/100);
 sal_final = sal_fixo+ comiss;
 System.out.printf("A sua comissão é ",comiss);
 System.out.printf("E seu Sálario FInal é ",sal_final);
  

    }
    
}
